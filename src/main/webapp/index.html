    <meta name="viewport" content="width=device-width, initial-scale=1.0">

    <link rel="stylesheet" href="styles.css"> <!-- Link to your CSS file -->

    <title>Traditions Quiz</title>

</head>

<body>

    <div id="content-container">

        <div id="fun-fact">Welcome to the Traditions Quiz! Click 'Get Tradition' to start.</div> <!-- Fun fact displayed here -->

        <div id="choice">

            <button class="a" onclick="submitGuess('Brazil')">Brazil</button>

            <button class="b" onclick="submitGuess('Germany')">Germany</button>

        </div>

        <div id="result"></div> <!-- Result message displayed here -->

        <button onclick="getTradition()">Get Tradition</button> <!-- Button to fetch a new tradition -->

    </div>


    <script>

        let correctCountry;


        function getTradition() {

            fetch('/api/tradition')

                .then(response => response.json())

                .then(data => {

                    document.getElementById('fun-fact').innerText = data.tradition;

                    correctCountry = data.country; // Store the correct country for validation

                    document.getElementById('result').innerText = ''; // Clear previous result

                });

        }


        function submitGuess(guess) {

            fetch(`/api/guess?guess=${guess}&correctCountry=${correctCountry}`)

                .then(response => response.text())

                .then(result => {

                    document.getElementById('result').innerText = result; // Display result

                });

        }


        // Load a tradition on page load

        window.onload = getTradition;

    </script>

</body>
